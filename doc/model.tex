\begin{equation*}
\dot{x_i} = x_i\Big(\Big(\sum_{j=1}^{n}a_{ij}x_j \Big)- \Big(\sum_{j=1}^{n}x_j\sum_{k=1}^{n}a_{jk}x_k\Big)\Big)
\end{equation*}

\begin{equation*}
\sum_{i=1}^n x_i = 1
\end{equation*}

For $n=3$ gives the hypercycle.

\begin{equation*}
A=
\begin{pmatrix}
0 & 1 & 0 \\
0 & 0 & 1 \\
1 & 0 & 0 \\
\end{pmatrix}
\end{equation*}
